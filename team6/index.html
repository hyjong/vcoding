quiz = [ ("H", "수소"), ("O", "산소"), ("C", "탄소"), ("N", "질소"), ("Na", "나트륨"), ("K", "칼륨"), ("Fe", "철"), ("Cu", "구리"), ] score = 0 print("🧪 원소 기호 퀴즈 시작!\n") for symbol, name in quiz: answer = input(f"{name}의 원소 기호는? ") if answer.strip().lower() == symbol.lower(): print("✔ 정답!\n") score += 1 else: print(f"✘ 오답! 정답은 {symbol}\n") print(f"🎯 최종 점수: {score}/{len(quiz)}")
